Pattern Matching 
fn classify(n: int) -> str {
    return match n {
        0 => "zero"
        1 => "one"
        x if x < 0 => "negative"
        _ => "other"
    }
}
Async/Await 
async fn fetch(url: str) {
    let res = await http_get(url)?
    return Ok(res.body)
}

async fn main() {
    match await fetch("https://api.example.com") {
        Ok(data) => println(data)
        Err(e) => println(e)
    }
}
HTTP Server 
import HTTP from "stdlib/http"

fn main() {
    let server = HTTP.Server(8080)
    
    server.get("/", fn(req, res) {
        res.body = "Hello!"
    })
    
    server.start()
}
